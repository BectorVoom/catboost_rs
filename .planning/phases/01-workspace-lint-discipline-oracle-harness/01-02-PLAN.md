  impl TFastRng64 {
    pub fn from_seed(seed: u64) -> Self;            // one-arg ctor (derives 4 args via TReallyFastRng32)
    pub fn new(seed1: u64, seq1: u32, seed2: u64, seq2: u32) -> Self;  // four-arg ctor + FixSeq
    pub fn gen_rand(&mut self) -> u64;              // (R1.mix(advance) << 32) | R2.mix(advance)
    pub fn uniform(&mut self, bound: u64) -> u64;   // rejection sampling; bound>0 (see error note)
    pub fn advance(&mut self, delta: u64);          // LcgAdvance on both streams
  }
